wine

    # Обновление пакетов и установка Wine

    # Скачивание и установка Python 3.11 для Windows через Wine
    wget https://www.python.org/ftp/python/3.11.0/python-3.11.0-amd64.exe
    wine python-3.11.0-amd64.exe /quiet InstallAllUsers=1 PrependPath=1

    # Установка Nuitka и зависимостей через Wine
    wine python -m pip install nuitka
    wine python -m pip install wheel
    wine python -m pip install setuptools
    wine python -m pip install scons

    # Установка всех компонентов MinGW-w64
    sudo pacman -S zenity mingw-w64-binutils mingw-w64-crt mingw-w64-gcc mingw-w64-headers mingw-w64-winpthreads

    # установка DLL
    sudo pacman -S winetricks
    sudo pacman -S cabextract
    winetriks mcf42
    cd где/лежит/mcf42.dll
    wine regsvr32 MFC42.dll

    # Компиляция скрипта в исполняемый файл
    wine python -m nuitka awards_v3.py --mingw64 --standalone --windows-disable-console --onefile --enable-plugin=tk-inter --windows-icon-from-ico=icon.ico -j2

    # или так   
    wine python -m nuitka awards_v3.9.py --mingw64 --standalone --windows-console-mode=disable --onefile --enable-plugin=tk-inter --windows-icon-from-ico=icon.ico -j2
